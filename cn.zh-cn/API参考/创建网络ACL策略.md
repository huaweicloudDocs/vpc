```

{
    "firewall_policy": {
        "name": "test-policy", 
        "firewall_rules": [
            "b8243448-cb3c-496e-851c-dadade4c161b"
        ]
    }
}
```

```

    "firewall_policy": {
        "description": "",

        "audited": false,

    }
}
```
